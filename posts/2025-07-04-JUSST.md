window.mewmew = `MASUKKAN_KODE_BASE64_ANDA_DI_SINI`;

function getRealContent(){
  try {
    let decoded = atob(window.mewmew);
    decoded = decoded.replace(/&lt;/g,'<').replace(/&gt;/g,'>').replace(/&quot;/g,'"').replace(/&#39;/g,"'");
    return decoded;
  } catch(e) {
    return "<p>Konten rusak.</p>";
  }
}

function tryUnlock(){
  const pass = document.getElementById("pass").value.trim();
  if (pass === "2026") {
    document.getElementById("lockScreen").style.display = "none";
    document.getElementById("contents").innerHTML = getRealContent();
  } else {
    alert("Password salah!");
  }
}

// Auto check (opsional)
if (typeof window.checkPreviousAccess === "function" && window.checkPreviousAccess()) {
  document.getElementById("contents").innerHTML = getRealContent();
} else {
  document.getElementById("lockScreen").style.display = "flex";
